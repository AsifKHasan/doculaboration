```{=latex}
% "top" strut
\newcommand\Tstrut[1]{\rule[2.0ex]{0pt}{#1}}

% "bottom" strut
\newcommand\Bstrut[1]{\rule[-0.9ex]{0pt}{#1}}

% Title Style - used for Level 0 heading
\newcommand\titlestyle[1]{{\vspace{0.2cm}\begin{center}\Large\scshape\textbf #1 \end{center}\vspace{0.2cm}}}

% vertical border macro [color and width]
\newcommand\vborder[2]{\color[RGB]{#1}\setlength\arrayrulewidth{#2}\vline}

% horizontal border macro [color and width]
\newcommand\hborder[2]{\arrayrulecolor[RGB]{#1}\global\deflength\arrayrulewidth{#2}}

% we set default table column separation to 0.05 inches
\setlength{\tabcolsep}{0.05in}

% we set default table row height to be 1.2 times of normal height
\renewcommand{\arraystretch}{1.2}

% we need a little bit space top of each row so that texts do not touch top border
\renewcommand\extrarowheight{2pt}

\newcommand\mc\multicolumn
\newcommand\mr\multirow

\setlength{\aboverulesep}{0pt}
\setlength{\belowrulesep}{0pt}
```
